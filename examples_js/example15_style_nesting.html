funcyStyle( '#header', {
              textAlign: 'center',
              h1: { fontSize_px: 26,
                    fontWeight: 'bold'
              },
              p: { fontSize_px: 12,
                   a: { textDecoration: 'none',
                     '&:hover': { borderWidth_px: 1 }
                   }
              }
            }
          );

}
